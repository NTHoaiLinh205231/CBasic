#INCLUDE <STDIO.H>
INT MAIN(INT ARGC, CHAR CONST *ARGV[])
{
	 *INP = FOPEN("INPUT.TXT", "R");
	IF (!INP) {
		PRINTF("OI MO TEP INPUT.TXT.\N");
		RETURN 1;
	}
	 *OUT = FOPEN("OUTPUT.TXT", "W");
	IF (!OUT) {
		PRINTF("OI MO TEP OUTPUT.TXT.\N");
		RETURN 1;
	}
	INT C;
	WHILE ((C=FGETC(INP)) != ) {
		FPUTC(C, OUT);
	}
	FCLOSE(INP);
	FCLOSE(OUT);
	RETURN 0;
}
